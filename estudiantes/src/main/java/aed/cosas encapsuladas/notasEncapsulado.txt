public double[] notas(){
    double[] res = new double[estudiantes.length];                                  // O(E)
    for (int i = 0; i < estudiantes.length; i++){                                   // O(E)
        res[i] = estudiantes[i].obtenerEstudiante().obtenerNota();                 // O(1)
    }
    return res;
} // O(E)
